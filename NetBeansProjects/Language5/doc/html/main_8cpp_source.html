26 int posMinMaxLanguage(const Language& language, const Language array[],
27  int initialPos, int finalPos, char operation) {
28  int posMinMax=-1;

31  if(finalPos>=initialPos) {

33  posMinMax = initialPos;
34  for(int i=initialPos; i<=finalPos; ++i) {

39  minMaxDistance = distance;

43  if (distance > minMaxDistance) {
44  minMaxDistance = distance;

47  }
